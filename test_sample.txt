//bubble sort
int a[10] , n;
void main() {
    int i , j , x;
    scanf(n);
    for (i = 0 ; i < n ; i = i + 1) {
        scanf(x);
        a[i] = x;
    }
    for (i = 0 ; i < n ; i = i + 1) {
        for (j = 0 ; j + 1 < n - i ; j = j + 1) {
            if (a[j] > a[j + 1]) {
                x = a[j];
                a[j] = a[j + 1];
                a[j + 1] = x;
            }
        }
    }
    for (i = 0 ; i < n ; i = i + 1) {
        printf(a[i]);
        if (i + 1 == n)
            printf("\n");
        else
            printf(" ");
    }
}
